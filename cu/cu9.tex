		\UCitem{Propósito}{Hacer que la apertura de cajas sea lo más eficiente posible para así reducir el tiempo que cada sucursal tarda en entrar en operaciones en el día  adía.

		\UCitem{Salidas}{Mensaje de confirmación de operación \bf {+ MSG0}.)}
		\UCitem{Destino}{Pantalla}
		\UCitem{Precondiciones}{La caja abrir debe estar cerrada.}
		\UCitem{Postcondiciones}{La caja quedará habilitada para el uso del cajero.}
		\UCitem{Errores}{Que no se realice una conexión a la base de datos,que el servidor se caiga.}

%--------------------------------------
	\begin{UCtrayectoria}{Principal}
		\UCpaso [\UCactor] Presiona el botón \IUbutton{Abrir Caja} que esta en la pantalla principal \IUref{IU1}{Pantalla principal}
		\UCpaso Verifica que los permisos de usuario sean permisos de supervisor. \Trayref{A}
		\UCpaso Despliega la \IUref {IU21} {Abrir Caja}.
		\UCpaso [\UCactor] Llena los campos requeridos en pantalla. 
		\UCpaso [\UCactor] Da clic en el botón.\IUbutton{Aceptar}.
		\UCpaso Habilita el uso de la caja.\Trayref{B}
		\UCpaso Despliega el mensaje \bf {+ MSG0}. 
		\UCpaso[\UCactor] Da clic en el \Ibutton {+ Ok }.
		\UCpaso Re-direcciona al \UCactor a la \IUref {IU13} {Tabla de Supervisores}.
		\UCpaso[\UCactor] Cierra su sesión.
	\end{UCtrayectoria}
%--------------------------------------		
	\begin{UCtrayectoriaA}{A}{Permiso Denegado}
			\UCpaso Muestra el Mensaje {\bf MSG4-}`"Cancelado[{\em Permiso Denegado }].".
			\UCpaso Muestra la pantalla \IUref{IU1}{Pantalla Principal}.
		\end{UCtrayectoriaA}
%--------------------------------------

		\begin{UCtrayectoriaA}{B}{La caja no puede ser habilitada}
			\UCpaso Despliega el mensaje \bf {+ MSG1 }.
			\UCpaso[\UCactor] Da clic en el \Ibutton {+ Ok }.
			\UCpaso Continua en el paso 1 del \UCref{CU9}.
		\end{UCtrayectoriaA}
