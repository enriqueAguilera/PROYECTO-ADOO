\begin{UseCase}{CU13}{Consultar Proveedor}{
		La farmacia consta con múltiples proveedores que nos surten medicamentos,  y para poder checar los datos de todos los proveedores que nos venden medicamentos necesitamos un listado de los proveedores con sus datos correspondientes.
	}
		\UCitem{Versión}{\color{Gray}0.2}
		\UCitem{Autor}{\color{Gray}Aguilera Rosas Landa Enrique}
		\UCitem{Supervisa}{\color{Gray}Felipe Zamora Gachuz}
		\UCitem{Actor}{Dueño,Supervisor}
		\UCitem{Propósito}{Consulta rápido y eficaz sobre los múltiples proveedores de la Farmacia.}
		\UCitem{Entradas}{Nombre del provedor, datos de factura}
		\UCitem{Origen}{Teclado}
		\UCitem{Salidas}{Todos los datos completo del proveedor}
		\UCitem{Destino}{Pantalla}
		\UCitem{Precondiciones}{El proveedor debe de estar registrado en el sistema.}
		\UCitem{Postcondiciones}{El dueño revisara la lista de los proveedores y sus respectivos datos .}
		\UCitem{Errores}{La pagina sea inaccesible por el momento debido a fallas con los servidores, proveedor no encontrado,}
		\UCitem{Observaciones}{}
		\UCitem{Estado}{Revisión}
	\end{UseCase}
